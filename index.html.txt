      let currentLang = 'en'; // 'en' or 'ar'
      let darkMode = false;

      const appRoot = document.getElementById('appRoot');

      function renderMenu() {
        const data = menuData[currentLang];
        const isRtl = currentLang === 'ar';
        const directionClass = isRtl ? 'rtl-direction' : 'ltr-direction';
        const fontClass = isRtl ? 'arabic-font' : 'english-font';

        let html = `
          <div class="${fontClass} ${directionClass}">
            <div class="header">
              <h1 class="restaurant-name">${data.restaurant}</h1>
              <div class="lang-theme-controls">
                <button class="ctrl-btn ${currentLang === 'en' ? 'active' : ''}" id="langEn">EN</button>
                <button class="ctrl-btn ${currentLang === 'ar' ? 'active' : ''}" id="langAr">AR</button>
                <button class="ctrl-btn" id="themeToggle">
                  <span class="theme-icon">${darkMode ? '☀️' : '🌙'}</span>
                </button>
              </div>
            </div>
            <div class="contact-line ${isRtl ? 'flex-reverse' : ''}">
              <span>📞 ${data.phone}</span>
              <span>✨ ${data.tagline}</span>
            </div>
            <div class="menu-grid">
        `;

        for (const [key, section] of Object.entries(data.sections)) {
          html += `<div class="section">`;
          html += `<div class="section-title">${section.title}</div>`;
          section.items.forEach(item => {
            html += `
              <div class="menu-item ${isRtl ? 'flex-reverse' : ''}">
                <div class="item-desc">
                  <span>${item.name}</span>
                  ${item.desc ? `<span class="item-sub">${item.desc}</span>` : ''}
                </div>
                <span class="price">${item.price}</span>
              </div>
            `;
          });
          html += `</div>`;
        }

        html += `
            </div>
            <footer>© Argana Brunch · Instagram @argana_Brunch</footer>
          </div>
        `;

        appRoot.innerHTML = html;

        // Attach event listeners
        document.getElementById('langEn')?.addEventListener('click', () => {
          if (currentLang !== 'en') {
            currentLang = 'en';
            renderMenu();
          }
        });
        document.getElementById('langAr')?.addEventListener('click', () => {
          if (currentLang !== 'ar') {
            currentLang = 'ar';
            renderMenu();
          }
        });
        document.getElementById('themeToggle')?.addEventListener('click', () => {
          darkMode = !darkMode;
          document.body.classList.toggle('dark', darkMode);
          // update theme icon inside button
          const themeBtn = document.getElementById('themeToggle');
          if (themeBtn) {
            themeBtn.innerHTML = `<span class="theme-icon">${darkMode ? '☀️' : '🌙'}</span>`;
          }
        });
      }

      // Initial render
      document.body.classList.remove('dark');
      renderMenu();

      // Ensure dark class is synced on load
      if (darkMode) {
        document.body.classList.add('dark');
      }
    })();
